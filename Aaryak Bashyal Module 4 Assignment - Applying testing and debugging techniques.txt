// more on two tests.
start
Declarations
num firstTest
num secondTest
num average
num PASSING = 60
output "Enter first score or 0 to quit "
while firstTest not equal to 0
output "Enter second score "
input secondTest
average = firstTest + secondTest / 2
ouput "Average is ", average
if average >= PASSING then
output "Pass"
else
output "Fail"
endif
output "Enter first score or 0 to quit "
endwhile
stop
